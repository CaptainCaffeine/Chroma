cmake_minimum_required(VERSION 2.8.12)
project(chroma-emu)

if(MSVC)
    set(STD_FLAG "/std:c++14")
elseif(("${CMAKE_CXX_COMPILER_ID}" STREQUAL "Clang") AND (CMAKE_CXX_COMPILER_VERSION VERSION_LESS 3.5))
    set(STD_FLAG "-std=c++1y")
else()
    set(STD_FLAG "-std=c++14")
endif()

if(MSVC)
    set(WARNING_FLAGS "/W4")
else()
    set(WARNING_FLAGS "-Wall -Wextra -Wshadow")
endif()

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${STD_FLAG} ${WARNING_FLAGS}")
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} ${STD_FLAG} ${WARNING_FLAGS}")
set(CMAKE_CXX_FLAGS_RELWITHDEBINFO "${CMAKE_CXX_FLAGS_RELWITHDEBINFO} ${STD_FLAG} ${WARNING_FLAGS}")
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} ${STD_FLAG}")
set(CMAKE_CXX_FLAGS_MINSIZEREL "${CMAKE_CXX_FLAGS_MINSIZEREL} ${STD_FLAG}")

set(CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake)

find_package(SDL2 REQUIRED)
include_directories(${SDL2_INCLUDE_DIR})

add_subdirectory(src)
